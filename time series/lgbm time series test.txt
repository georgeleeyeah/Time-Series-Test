"""
Production LightGBM 12-Month-Sum Forecast Pipeline
====================================================
Input:  DataFrame with columns [tin, ds, y] — all historical data (no OOT)
Output: DataFrame with columns [tin, origin_date, forecast_12m_sum]

The pipeline:
  1. Uses ALL input data for training (sliding window generates train rows)
  2. Internal train/val split is ONLY for early stopping + feature selection
  3. Final model is trained on ALL training rows
  4. Produces one forecast per tin using all available history
  5. You evaluate externally by comparing forecast_12m_sum to your OOT actuals

Usage:
    from production_pipeline import run_pipeline
    
    df = pd.read_csv("my_data.csv")  # all data before 2025-03
    
    forecasts = run_pipeline(
        df,
        min_history_months=12,
        max_origins_per_tin=6,
        output_dir="./output",
    )
    
    # forecasts: tin | origin_date | forecast_12m_sum
    # You pull in OOT actuals (2025-03 to 2026-03) and compute median APE yourself
"""

import numpy as np
import pandas as pd
import lightgbm as lgb
from typing import Optional
import warnings

warnings.filterwarnings("ignore")

from lgbm_12m_forecast_pipeline import (
    prepare_data,
    fill_missing_months,
    make_features_for_tin,
    select_features,
    tune_hyperparameters,
    show_feature_importance,
)

# Override: these columns are NOT features
NON_FEATURE_COLS = {"tin", "origin_date", "target", "target_ratio", "base_12m"}


def get_feature_cols(df: pd.DataFrame) -> list:
    return [c for c in df.columns if c not in NON_FEATURE_COLS]


# =============================================================================
# TRAINING DATA: sliding window
# =============================================================================

def build_training_data(
    df: pd.DataFrame,
    min_history_months: int = 12,
    forecast_horizon: int = 12,
    max_origins_per_tin: Optional[int] = 6,
) -> pd.DataFrame:
    """
    Build training rows using sliding window.
    
    Stores two targets:
      - target: raw 12-month sum (for reference)
      - target_ratio: next_12m_sum / trailing_12m_sum (scale-free, what we actually model)
      - base_12m: trailing 12-month sum at the origin (used to convert ratio back to $)
    
    Training on target_ratio means a $100/month tin and a $10,000/month tin
    contribute equally to the loss. The model learns growth/decline patterns
    rather than memorizing dollar levels.
    """
    all_rows = []
    for tin, grp in df.groupby("tin"):
        grp = grp.sort_values("ds").reset_index(drop=True)
        cashflows = grp["y"].values
        dates = grp["ds"].values
        n = len(cashflows)

        first_origin = min_history_months - 1
        last_origin = n - forecast_horizon - 1
        if first_origin > last_origin:
            continue

        origins = list(range(first_origin, last_origin + 1))
        if max_origins_per_tin and len(origins) > max_origins_per_tin:
            origins = origins[-max_origins_per_tin:]

        for oidx in origins:
            target_raw = cashflows[oidx + 1: oidx + 1 + forecast_horizon].sum()
            origin_date = pd.Timestamp(dates[oidx])
            feats = make_features_for_tin(cashflows, oidx, origin_date)

            # Trailing 12m sum as the base for ratio
            history = cashflows[:oidx + 1]
            trailing_12m = history[-min(12, len(history)):].sum()

            feats["tin"] = tin
            feats["origin_date"] = origin_date
            feats["target"] = target_raw
            feats["base_12m"] = trailing_12m
            # Ratio target: next 12m / trailing 12m (scale-free)
            feats["target_ratio"] = target_raw / (trailing_12m + 1e-9)
            all_rows.append(feats)

    return pd.DataFrame(all_rows)


# =============================================================================
# INFERENCE DATA: one row per tin using all history
# =============================================================================

def build_inference_data(df: pd.DataFrame) -> pd.DataFrame:
    """
    One row per tin. Origin = last month. Features from all history.
    Includes base_12m (trailing 12m sum) for converting ratio prediction back to $.
    """
    all_rows = []
    for tin, grp in df.groupby("tin"):
        grp = grp.sort_values("ds").reset_index(drop=True)
        cashflows = grp["y"].values
        dates = grp["ds"].values
        origin_idx = len(cashflows) - 1
        origin_date = pd.Timestamp(dates[origin_idx])
        feats = make_features_for_tin(cashflows, origin_idx, origin_date)
        feats["tin"] = tin
        feats["origin_date"] = origin_date
        # Trailing 12m sum — used to convert ratio back to dollars
        feats["base_12m"] = cashflows[-min(12, len(cashflows)):].sum()
        all_rows.append(feats)
    return pd.DataFrame(all_rows)


# =============================================================================
# TRAIN / VAL SPLIT (internal only — for early stopping & feature selection)
# =============================================================================

def internal_train_val_split(train_data: pd.DataFrame, val_frac: float = 0.3) -> tuple:
    """
    Split training rows 70/30 by origin date.
    This is internal only — both sets come from the same historical data.
    The val set is for early stopping, NOT for final evaluation.
    """
    sorted_dates = sorted(train_data["origin_date"].unique())
    split_idx = max(1, int(len(sorted_dates) * (1 - val_frac)))
    cutoff = sorted_dates[split_idx - 1]
    train_set = train_data[train_data["origin_date"] <= cutoff].copy()
    val_set = train_data[train_data["origin_date"] > cutoff].copy()

    # Safety: if either is empty, force a 50/50 split
    if len(train_set) == 0 or len(val_set) == 0:
        mid = max(1, len(sorted_dates) // 2)
        cutoff = sorted_dates[mid - 1]
        train_set = train_data[train_data["origin_date"] <= cutoff].copy()
        val_set = train_data[train_data["origin_date"] > cutoff].copy()

    print(f"  Train: {len(train_set)} rows ({train_set['tin'].nunique()} tins), "
          f"origins up to {cutoff.date()}")
    print(f"  Val:   {len(val_set)} rows ({val_set['tin'].nunique()} tins), "
          f"origins after {cutoff.date()}")
    print(f"  (Internal split for early stopping only — not for final evaluation)")
    return train_set, val_set


# =============================================================================
# MODEL TRAINING
# =============================================================================

def train_model(
    train_set: pd.DataFrame,
    val_set: pd.DataFrame,
    feature_cols: list,
    lgb_params: Optional[dict] = None,
    categorical_features: Optional[list] = None,
    retrain_on_all: bool = True,
) -> lgb.Booster:
    """
    Train LightGBM on target_ratio (next_12m / trailing_12m).
    
    Training on the ratio makes the model scale-free:
    a $100/month tin and a $10,000/month tin contribute equally.
    
    If retrain_on_all=True (default), first finds best iteration using
    train/val split, then retrains on ALL data with that iteration count.
    """
    if lgb_params is None:
        lgb_params = {
            "objective": "regression",
            "metric": "mae",
            "learning_rate": 0.05,
            "num_leaves": 63,
            "max_depth": 6,
            "min_child_samples": 30,
            "feature_fraction": 0.7,
            "bagging_fraction": 0.8,
            "bagging_freq": 5,
            "reg_alpha": 0.1,
            "reg_lambda": 1.0,
            "verbosity": -1,
            "n_jobs": -1,
            "seed": 42,
        }

    if categorical_features is None:
        categorical_features = ["origin_month", "origin_quarter", "tenure_bucket"]
    cat_feats = [c for c in categorical_features if c in feature_cols]

    # Train on target_ratio (scale-free)
    dtrain = lgb.Dataset(
        train_set[feature_cols], label=train_set["target_ratio"],
        categorical_feature=cat_feats, free_raw_data=False,
    )
    dval = lgb.Dataset(
        val_set[feature_cols], label=val_set["target_ratio"],
        categorical_feature=cat_feats, reference=dtrain, free_raw_data=False,
    )

    model = lgb.train(
        lgb_params, dtrain, num_boost_round=3000,
        valid_sets=[dtrain, dval], valid_names=["train", "val"],
        callbacks=[lgb.early_stopping(50), lgb.log_evaluation(200)],
    )
    best_iter = model.best_iteration
    print(f"\n  Best iteration (from val): {best_iter}")

    if retrain_on_all:
        print(f"  Retraining on ALL {len(train_set) + len(val_set)} rows "
              f"with {best_iter} iterations...")
        all_data = pd.concat([train_set, val_set], ignore_index=True)
        dall = lgb.Dataset(
            all_data[feature_cols], label=all_data["target_ratio"],
            categorical_feature=cat_feats, free_raw_data=False,
        )
        model = lgb.train(
            lgb_params, dall, num_boost_round=best_iter,
            callbacks=[lgb.log_evaluation(0)],
        )
        print(f"  Final model trained on {len(all_data)} rows.")

    return model


# =============================================================================
# PIPELINE
# =============================================================================

def run_pipeline(
    df: pd.DataFrame,
    min_history_months: int = 12,
    max_origins_per_tin: Optional[int] = 6,
    lgb_params: Optional[dict] = None,
    # Feature selection
    do_feature_selection: bool = True,
    corr_threshold: float = 0.95,
    min_importance_pct: float = 0.001,
    top_n_features: Optional[int] = None,
    # Tuning
    do_tune: bool = False,
    tune_n_trials: int = 50,
    tune_cv_folds: int = 3,
    # Output
    output_dir: Optional[str] = None,
) -> pd.DataFrame:
    """
    Production pipeline: train on ALL data, forecast next 12m for each tin.

    Parameters
    ----------
    df : all historical data [tin, ds, y] — everything before your OOT cutoff
    min_history_months : minimum history months at each origin (default 12)
    max_origins_per_tin : training rows per tin (default 6, controls memory)
    lgb_params : LightGBM params (None = use defaults or tuned)
    do_feature_selection : run multi-stage feature selection (default True)
    do_tune : run hyperparameter tuning (default False)
    output_dir : save all artifacts to this dir (default None = don't save)

    Returns
    -------
    pd.DataFrame: tin | origin_date | forecast_12m_sum
    """
    import pickle
    import os

    def _save(obj, filename):
        if output_dir is None:
            return
        os.makedirs(output_dir, exist_ok=True)
        path = os.path.join(output_dir, filename)
        if filename.endswith(".txt"):
            obj.save_model(path)
        elif filename.endswith(".pkl"):
            with open(path, "wb") as f:
                pickle.dump(obj, f, protocol=pickle.HIGHEST_PROTOCOL)
        print(f"    -> Saved: {path}")

    if output_dir:
        print(f"  Artifacts will be saved to: {output_dir}/\n")

    # ── STEP 1: Prepare ──
    print("=" * 60)
    print("STEP 1: Preparing data")
    print("=" * 60)
    df = prepare_data(df)
    df = fill_missing_months(df)
    print(f"  {df['tin'].nunique()} tins, {len(df)} rows")
    print(f"  Date range: {df['ds'].min().date()} to {df['ds'].max().date()}")
    _save(df, "01_prepared_data.pkl")

    # ── STEP 2: Build training rows ──
    print("\n" + "=" * 60)
    print("STEP 2: Building training rows (sliding window)")
    print("=" * 60)
    train_data = build_training_data(
        df, min_history_months=min_history_months,
        forecast_horizon=12, max_origins_per_tin=max_origins_per_tin,
    )
    print(f"  {len(train_data)} rows from {train_data['tin'].nunique()} tins")
    print(f"  {len(get_feature_cols(train_data))} features")
    _save(train_data, "02_train_data.pkl")

    if len(train_data) == 0:
        raise ValueError("No training rows. Try lowering min_history_months.")

    # ── STEP 3: Internal train/val split ──
    print("\n" + "=" * 60)
    print("STEP 3: Internal train/val split (for early stopping only)")
    print("=" * 60)
    train_set, val_set = internal_train_val_split(train_data)
    _save(train_set, "03_train_set.pkl")
    _save(val_set, "04_val_set.pkl")

    # ── STEP 4: Feature selection ──
    # select_features internally uses the 'target' column, so we temporarily
    # swap it to target_ratio so feature selection is also scale-free
    selected_features = get_feature_cols(train_set)
    selection_report = None
    if do_feature_selection:
        print("\n" + "=" * 60)
        print("STEP 4: Feature Selection")
        print("=" * 60)
        # Drop columns that shouldn't be seen, rename ratio→target
        drop_cols = ["target", "base_12m"]
        keep_cols = [c for c in train_set.columns if c not in drop_cols]
        train_sel = train_set[keep_cols].rename(columns={"target_ratio": "target"})
        val_sel = val_set[keep_cols].rename(columns={"target_ratio": "target"})
        selected_features, selection_report = select_features(
            train_sel, val_sel,
            corr_threshold=corr_threshold,
            min_importance_pct=min_importance_pct,
            top_n=top_n_features,
            lgb_params=lgb_params,
        )
        _save(selection_report, "05_selection_report.pkl")
    _save(selected_features, "06_selected_features.pkl")

    # ── STEP 5: Tuning (optional) ──
    tuning_log = None
    if do_tune and lgb_params is None:
        print("\n" + "=" * 60)
        print(f"STEP 5: Hyperparameter Tuning ({tune_n_trials} trials)")
        print("=" * 60)
        lgb_params, tuning_log = tune_hyperparameters(
            train_set, selected_features,
            n_folds=tune_cv_folds,
            n_trials=tune_n_trials,
        )
        _save(tuning_log, "07_tuning_log.pkl")
        _save(lgb_params, "07b_best_params.pkl")

    # ── STEP 6: Train final model on ALL training rows ──
    print("\n" + "=" * 60)
    print(f"STEP 6: Training final model ({len(selected_features)} features)")
    print("=" * 60)
    model = train_model(
        train_set, val_set,
        feature_cols=selected_features,
        lgb_params=lgb_params,
        retrain_on_all=True,  # uses val to find best iter, then retrains on everything
    )
    _save(model, "08_lgbm_model.txt")

    # ── STEP 7: Feature importance ──
    print("\n" + "=" * 60)
    print("STEP 7: Feature Importance")
    print("=" * 60)
    imp_df = show_feature_importance(model)
    _save(imp_df, "09_feature_importance.pkl")

    # ── STEP 8: Forecast ──
    print("\n" + "=" * 60)
    print("STEP 8: Generating forecasts (one per tin, using all history)")
    print("=" * 60)
    inference_data = build_inference_data(df)

    # Model predicts the RATIO (next_12m / trailing_12m)
    predicted_ratio = model.predict(inference_data[selected_features])

    # Convert back to dollars: ratio × base_12m
    inference_data["predicted_ratio"] = predicted_ratio
    inference_data["forecast_12m_sum"] = predicted_ratio * inference_data["base_12m"]

    forecasts = inference_data[["tin", "origin_date", "base_12m",
                                "predicted_ratio", "forecast_12m_sum"]].copy()
    print(f"  {len(forecasts)} tins forecasted")
    print(f"  Predicted ratio — min: {predicted_ratio.min():.3f}, "
          f"median: {np.median(predicted_ratio):.3f}, "
          f"max: {predicted_ratio.max():.3f}")
    print(f"  Forecast 12m sum — min: {forecasts['forecast_12m_sum'].min():,.0f}, "
          f"median: {forecasts['forecast_12m_sum'].median():,.0f}, "
          f"max: {forecasts['forecast_12m_sum'].max():,.0f}")
    _save(inference_data, "10_inference_features.pkl")
    _save(forecasts, "11_forecasts.pkl")

    if output_dir:
        print(f"\n  All artifacts saved to: {output_dir}/")

    run_pipeline.last_results = {
        "model": model,
        "selected_features": selected_features,
        "selection_report": selection_report,
        "tuning_log": tuning_log,
        "lgb_params_used": lgb_params,
        "feature_importance": imp_df,
        "inference_features": inference_data,
    }

    return forecasts

        base = np.random.uniform(1000, 50000)
        trend = np.random.uniform(-50, 100)
        values = [
            max(base + trend * t + np.random.normal(0, base * 0.1), 0)
            for t in range(tenure)
        ]
        for d, v in zip(dates, values):
            rows.append({"tin": f"TIN_{i:04d}", "ds": d, "y": v})

    df = pd.DataFrame(rows)
    forecasts = run_pipeline(

        min_history_months=12,

    )
